188 // Read A .RAW File In To The Allocated Image Buffer Using Data In The Image Structure Header.
189 // Flip The Image Top To Bottom. Returns 0 For Failure Of Read, Or Number Of Bytes Read.
190 int ReadTextureData ( char *filename, P_TEXTURE_IMAGE buffer)
191 {
192  FILE *f;
193  int i,j,k,done=0;
194  int stride = buffer->width * buffer->format; // Size Of A Row (Width * Bytes Per Pixel)
195  unsigned char *p = NULL;
196 
197  f = fopen(filename, "rb"); // Open "filename" For Reading Bytes
198  if( f != NULL ) // If File Exists
199  {
200  for( i = buffer->height-1; i >= 0 ; i-- ) // Loop Through Height (Bottoms Up - Flip Image)
201  {
202  p = buffer->data + (i * stride ); //
203  for ( j = 0; j < buffer->width ; j++ ) // Loop Through Width
204  {
205  for ( k = 0 ; k < buffer->format-1 ; k++, p++, done++ )
206  {
207  *p = fgetc(f); // Read Value From File And Store In Memory
208  }
209  *p = 255; p++; // Store 255 In Alpha Channel And Increase Pointer
210  }
211  }
212  fclose(f); // Close The File
213  }
214  else // Otherwise
215  {
216  MessageBox(NULL,"Unable To Open Image File","IMAGE ERROR",MB_OK | MB_ICONINFORMATION);
217  }
218  return done; // Returns Number Of Bytes Read In
219 }

230 void Blit( P_TEXTURE_IMAGE src, P_TEXTURE_IMAGE dst, int src_xstart, int src_ystart, int src_width, int src_height,
231  int dst_xstart, int dst_ystart, int blend, int alpha)
232 {
233  int i,j,k;
234  unsigned char *s, *d; // Source & Destination
235 
236  // Clamp Alpha If Value Is Out Of Range
237  if( alpha > 255 ) alpha = 255;
238  if( alpha < 0 ) alpha = 0;
239 
240  // Check For Incorrect Blend Flag Values
241  if( blend < 0 ) blend = 0;
242  if( blend > 1 ) blend = 1;
243 
244  d = dst->data + (dst_ystart * dst->width * dst->format); // Start Row - dst (Row * Width In Pixels * Bytes Per Pixel)
245  s = src->data + (src_ystart * src->width * src->format); // Start Row - src (Row * Width In Pixels * Bytes Per Pixel)
246 
247  for (i = 0 ; i < src_height ; i++ ) // Height Loop
248  {
249  s = s + (src_xstart * src->format); // Move Through Src Data By Bytes Per Pixel
250  d = d + (dst_xstart * dst->format); // Move Through Dst Data By Bytes Per Pixel
251  for (j = 0 ; j < src_width ; j++ ) // Width Loop
252  {
253  for( k = 0 ; k < src->format ; k++, d++, s++) // "n" Bytes At A Time
254  {
255  if (blend) // If Blending Is On
256  *d = ( (*s * alpha) + (*d * (255-alpha)) ) >> 8; // Multiply Src Data*alpha Add Dst Data*(255-alpha)
257  else // Keep in 0-255 Range With >> 8
258  *d = *s; // No Blending Just Do A Straight Copy
259  }
260  }
261  d = d + (dst->width - (src_width + dst_xstart))*dst->format; // Add End Of Row */
262  s = s + (src->width - (src_width + src_xstart))*src->format; // Add End Of Row */
263  }
264 }

570  UINT uMsg, // Message For This Window
571  WPARAM wParam, // Additional Message Information
572  LPARAM lParam) // Additional Message Information
573 {
574  switch (uMsg) // Check For Windows Messages
575  {

577  {

579  {

581  }

583  {
584  active=FALSE; // Program Is No Longer Active
585  }

588  }

590  case WM_SYSCOMMAND: // Intercept System Commands
591  {
592  switch (wParam) // Check System Calls
